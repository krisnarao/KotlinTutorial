import { useEffect, useState } from "react";
import {
  Card,
  CardContent,
  Typography,
  Grid,
  TextField,
  MenuItem,
  Button,
  Alert,
  CircularProgress,
} from "@mui/material";

import serverService from "../../services/serverService";
import secretService from "../../services/secretService";

const durations = [15, 30, 60, 120];

const initialForm = {
  serverId: "",
  secretId: "",
  incidentId: "",
  duration: 30,
  justification: "",
  requestedBy: "admin",
};

function RequestForm() {
  const [form, setForm] = useState(initialForm);

  const [servers, setServers] = useState([]);
  const [allSecrets, setAllSecrets] = useState([]);
  const [filteredSecrets, setFilteredSecrets] = useState([]);

  const [loading, setLoading] = useState(true);

  useEffect(() => {
    loadData();
  }, []);

  const loadData = async () => {
    try {
      setLoading(true);

      const serverResponse = await serverService.getServerList();
      setServers(serverResponse);

      const secretResponse = await secretService.getAllSecrets();
      setAllSecrets(secretResponse);
    } catch (err) {
      console.error(err);
    } finally {
      setLoading(false);
    }
  };

  const handleChange = (event) => {
    const { name, value } = event.target;

    const updatedForm = {
      ...form,
      [name]: value,
    };

    if (name === "serverId") {
      const selectedServer = servers.find(
        (server) => server.id === Number(value)
      );

      if (selectedServer) {
        const secrets = allSecrets.filter(
          (secret) => secret.serverName === selectedServer.hostname
        );

        setFilteredSecrets(secrets);
      } else {
        setFilteredSecrets([]);
      }

      updatedForm.secretId = "";
    }

    setForm(updatedForm);
  };

  const handleSubmit = () => {
    if (
      !form.serverId ||
      !form.secretId ||
      !form.incidentId ||
      !form.justification
    ) {
      alert("Please fill all mandatory fields.");
      return;
    }

    console.log("Break Glass Request", form);

    // TODO:
    // breakGlassService.createRequest(form);
  };

  if (loading) {
    return (
      <Card sx={{ borderRadius: 4 }}>
        <CardContent
          sx={{
            display: "flex",
            justifyContent: "center",
            p: 5,
          }}
        >
          <CircularProgress />
        </CardContent>
      </Card>
    );
  }

  return (
    <Card sx={{ borderRadius: 4 }}>
      <CardContent>

        <Typography variant="h5" fontWeight="bold" gutterBottom>
          Emergency Break Glass Request
        </Typography>

        <Alert severity="warning" sx={{ mb: 3 }}>
          Emergency privileged access is time limited and fully audited.
        </Alert>

        <Grid container spacing={2}>

          <Grid item xs={12} md={6}>
            <TextField
              select
              fullWidth
              required
              label="Server"
              name="serverId"
              value={form.serverId}
              onChange={handleChange}
            >
              {servers.map((server) => (
                <MenuItem key={server.id} value={server.id}>
                  {server.hostname}
                </MenuItem>
              ))}
            </TextField>
          </Grid>

          <Grid item xs={12} md={6}>
            <TextField
              select
              fullWidth
              required
              disabled={!form.serverId}
              label="Secret"
              name="secretId"
              value={form.secretId}
              onChange={handleChange}
            >
              {filteredSecrets.map((secret) => (
                <MenuItem
                  key={secret.id}
                  value={secret.id}
                >
                  {secret.secretName}
                </MenuItem>
              ))}
            </TextField>
          </Grid>

          <Grid item xs={12} md={6}>
            <TextField
              fullWidth
              required
              label="Incident ID"
              name="incidentId"
              value={form.incidentId}
              onChange={handleChange}
              placeholder="INC-2026-001245"
            />
          </Grid>

          <Grid item xs={12} md={6}>
            <TextField
              select
              fullWidth
              label="Access Duration"
              name="duration"
              value={form.duration}
              onChange={handleChange}
            >
              {durations.map((duration) => (
                <MenuItem
                  key={duration}
                  value={duration}
                >
                  {duration} Minutes
                </MenuItem>
              ))}
            </TextField>
          </Grid>

          <Grid item xs={12}>
            <TextField
              fullWidth
              required
              multiline
              rows={4}
              label="Business Justification"
              name="justification"
              value={form.justification}
              onChange={handleChange}
            />
          </Grid>

          <Grid item xs={12}>
            <TextField
              fullWidth
              label="Requested By"
              value={form.requestedBy}
              InputProps={{
                readOnly: true,
              }}
            />
          </Grid>

          <Grid item xs={12}>
            <Button
              variant="contained"
              size="large"
              fullWidth
              sx={{ height: 50 }}
              onClick={handleSubmit}
            >
              Submit Break Glass Request
            </Button>
          </Grid>

        </Grid>

      </CardContent>
    </Card>
  );
}

export default RequestForm;
