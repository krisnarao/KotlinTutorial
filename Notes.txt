import api from "./api";

const API_URL = "/api/secrets";

const getAllSecrets = async () => {
    const response = await api.get(API_URL);
    return response.data;
};

const getSecretById = async (id) => {
    const response = await api.get(`${API_URL}/${id}`);
    return response.data;
};

const createSecret = async (secret) => {
    const response = await api.post(API_URL, secret);
    return response.data;
};

const updateSecret = async (id, secret) => {
    const response = await api.put(`${API_URL}/${id}`, secret);
    return response.data;
};

const deleteSecret = async (id) => {
    return await api.delete(`${API_URL}/${id}`);
};

export default {
    getAllSecrets,
    getSecretById,
    createSecret,
    updateSecret,
    deleteSecret
};

import { useEffect, useState } from "react";
import {
  Dialog,
  DialogTitle,
  DialogContent,
  DialogActions,
  Button,
  Grid,
  TextField,
  MenuItem,
  Switch,
  FormControlLabel,
} from "@mui/material";

const secretTypes = [
  "PASSWORD",
  "API_KEY",
  "CERTIFICATE",
  "TOKEN",
  "SSH_KEY",
];

const initialForm = {
  secretName: "",
  vaultReference: "",
  secretType: "",
  rotationDays: 90,
  serverId: "",
  active: true,
};

function SecretDialog({
  open,
  onClose,
  onSave,
  secret,
  servers = [],
}) {
  const [form, setForm] = useState(initialForm);

  useEffect(() => {
    if (secret) {
      setForm({
        secretName: secret.secretName || "",
        vaultReference: secret.vaultReference || "",
        secretType: secret.secretType || "",
        rotationDays: secret.rotationDays || 90,
        serverId: secret.serverId || "",
        active:
          secret.active === undefined ? true : secret.active,
      });
    } else {
      setForm(initialForm);
    }
  }, [secret]);

  const handleChange = (event) => {
    const { name, value } = event.target;

    setForm((prev) => ({
      ...prev,
      [name]: value,
    }));
  };

  const handleSwitch = (event) => {
    setForm((prev) => ({
      ...prev,
      active: event.target.checked,
    }));
  };

  const handleSubmit = () => {
    if (
      !form.secretName ||
      !form.vaultReference ||
      !form.secretType ||
      !form.serverId
    ) {
      alert("Please fill all mandatory fields.");
      return;
    }

    onSave(form);
  };

  return (
    <Dialog
      open={open}
      onClose={onClose}
      fullWidth
      maxWidth="md"
    >
      <DialogTitle>
        {secret ? "Edit Secret" : "Add Secret"}
      </DialogTitle>

      <DialogContent dividers>
        <Grid container spacing={2} sx={{ mt: 0.5 }}>

          <Grid item xs={12} md={6}>
            <TextField
              fullWidth
              required
              label="Secret Name"
              name="secretName"
              value={form.secretName}
              onChange={handleChange}
            />
          </Grid>

          <Grid item xs={12} md={6}>
            <TextField
              fullWidth
              required
              label="Vault Reference"
              name="vaultReference"
              value={form.vaultReference}
              onChange={handleChange}
            />
          </Grid>

          <Grid item xs={12} md={6}>
            <TextField
              select
              fullWidth
              required
              label="Secret Type"
              name="secretType"
              value={form.secretType}
              onChange={handleChange}
            >
              {secretTypes.map((type) => (
                <MenuItem
                  key={type}
                  value={type}
                >
                  {type}
                </MenuItem>
              ))}
            </TextField>
          </Grid>

          <Grid item xs={12} md={6}>
            <TextField
              fullWidth
              required
              type="number"
              label="Rotation Days"
              name="rotationDays"
              value={form.rotationDays}
              onChange={handleChange}
            />
          </Grid>

          <Grid item xs={12}>
            <TextField
              select
              fullWidth
              required
              label="Server"
              name="serverId"
              value={form.serverId}
              onChange={handleChange}
            >
              {servers.map((server) => (
                <MenuItem
                  key={server.id}
                  value={server.id}
                >
                  {server.hostname}
                </MenuItem>
              ))}
            </TextField>
          </Grid>

          {secret && (
            <>
              <Grid item xs={12} md={6}>
                <TextField
                  fullWidth
                  label="Last Rotated"
                  value={secret.lastRotated || ""}
                  InputProps={{
                    readOnly: true,
                  }}
                />
              </Grid>

              <Grid item xs={12} md={6}>
                <TextField
                  fullWidth
                  label="Expiry Date"
                  value={secret.expiryDate || ""}
                  InputProps={{
                    readOnly: true,
                  }}
                />
              </Grid>
            </>
          )}

          <Grid item xs={12}>
            <FormControlLabel
              control={
                <Switch
                  checked={form.active}
                  onChange={handleSwitch}
                />
              }
              label="Active"
            />
          </Grid>

        </Grid>
      </DialogContent>

      <DialogActions>
        <Button
          onClick={onClose}
          color="inherit"
        >
          Cancel
        </Button>

        <Button
          variant="contained"
          onClick={handleSubmit}
        >
          {secret ? "Update Secret" : "Save Secret"}
        </Button>
      </DialogActions>
    </Dialog>
  );
}

export default SecretDialog;

import {
  Paper,
  Table,
  TableHead,
  TableRow,
  TableCell,
  TableBody,
  Chip,
  IconButton,
  Tooltip,
  Typography,
} from "@mui/material";

import EditIcon from "@mui/icons-material/Edit";
import DeleteIcon from "@mui/icons-material/Delete";

function getStatus(secret) {
  if (!secret.active) {
    return {
      label: "Inactive",
      color: "default",
    };
  }

  if (!secret.expiryDate) {
    return {
      label: "Active",
      color: "success",
    };
  }

  const expiry = new Date(secret.expiryDate);
  const today = new Date();

  const diff = Math.ceil(
    (expiry - today) / (1000 * 60 * 60 * 24)
  );

  if (diff < 0) {
    return {
      label: "Expired",
      color: "error",
    };
  }

  if (diff <= 30) {
    return {
      label: "Expiring",
      color: "warning",
    };
  }

  return {
    label: "Active",
    color: "success",
  };
}

}) {

            <TableRow>

              <TableCell
                colSpan={9}

                  color="text.secondary"
                >
                  No secrets found

          )}

          {secrets.map((secret) => {

            const status = getStatus(secret);

import { useEffect, useState } from "react";
import {
  Box,
  Container,
  Typography,
  Stack,
  Button,
  TextField,
  CircularProgress,
  Snackbar,
  Alert,
} from "@mui/material";

import AddIcon from "@mui/icons-material/Add";
import RefreshIcon from "@mui/icons-material/Refresh";

import Sidebar from "../components/layout/Sidebar";
import Header from "../components/layout/Header";

import SecretTable from "../components/secret/SecretTable";
import SecretDialog from "../components/secret/SecretDialog";

import secretService from "../services/secretService";
import serverService from "../services/serverService";

function SecretInventory() {

  const [secrets, setSecrets] = useState([]);
  const [filteredSecrets, setFilteredSecrets] = useState([]);

  const [servers, setServers] = useState([]);

  const [loading, setLoading] = useState(true);

  const [search, setSearch] = useState("");

  const [dialogOpen, setDialogOpen] = useState(false);

  const [selectedSecret, setSelectedSecret] = useState(null);

  const [snackbar, setSnackbar] = useState({
    open: false,
    message: "",
    severity: "success",
  });

  useEffect(() => {
    loadData();
  }, []);

  useEffect(() => {

    if (!search) {

      setFilteredSecrets(secrets);

      return;

    }

    const filtered = secrets.filter((secret) =>
      secret.secretName.toLowerCase().includes(search.toLowerCase())
    );

    setFilteredSecrets(filtered);

  }, [search, secrets]);

  const loadData = async () => {

    try {

      setLoading(true);

      const secretResponse = await secretService.getAllSecrets();

      setSecrets(secretResponse);

      setFilteredSecrets(secretResponse);

      const serverResponse = await serverService.getServerList();

      setServers(serverResponse);

    } catch (err) {

      console.error(err);

      setSnackbar({
        open: true,
        message: "Failed to load secrets",
        severity: "error",
      });

    } finally {

      setLoading(false);

    }

  };

  const handleAdd = () => {

    setSelectedSecret(null);

    setDialogOpen(true);

  };

  const handleEdit = (secret) => {

    setSelectedSecret(secret);

    setDialogOpen(true);

  };

  const handleDelete = async (secret) => {

    if (!window.confirm(`Delete ${secret.secretName}?`)) return;

    try {

      await secretService.deleteSecret(secret.id);

      loadData();

      setSnackbar({
        open: true,
        message: "Secret deleted successfully",
        severity: "success",
      });

    } catch (err) {

      setSnackbar({
        open: true,
        message: "Delete failed",
        severity: "error",
      });

    }

  };

  const handleSave = async (form) => {

    try {

      if (selectedSecret) {

        await secretService.updateSecret(
          selectedSecret.id,
          form
        );

        setSnackbar({
          open: true,
          message: "Secret updated successfully",
          severity: "success",
        });

      } else {

        await secretService.createSecret(form);

        setSnackbar({
          open: true,
          message: "Secret created successfully",
          severity: "success",
        });

      }

      setDialogOpen(false);

      loadData();

    } catch (err) {

      console.error(err);

      setSnackbar({
        open: true,
        message: "Save failed",
        severity: "error",
      });

    }

  };

  return (

    <Box
      sx={{
        display: "flex",
        minHeight: "100vh",
        bgcolor: "#F5F7FB",
      }}
    >

      <Sidebar />

      <Box sx={{ flex: 1, p: 3 }}>

        <Header />

        <Container maxWidth="xl">

          <Typography
            variant="h4"
            fontWeight={700}
            mb={3}
          >
            Secret Inventory
          </Typography>

          <Stack
            direction="row"
            spacing={2}
            mb={3}
          >

            <TextField
              label="Search Secret"
              value={search}
              onChange={(e) =>
                setSearch(e.target.value)
              }
              sx={{ width: 300 }}
            />

            <Button
              variant="contained"
              startIcon={<AddIcon />}
              onClick={handleAdd}
            >
              Add Secret
            </Button>

            <Button
              variant="outlined"
              startIcon={<RefreshIcon />}
              onClick={loadData}
            >
              Refresh
            </Button>

          </Stack>

          {loading ? (

            <Box
              display="flex"
              justifyContent="center"
              mt={8}
            >
              <CircularProgress />
            </Box>

          ) : (

            <SecretTable
              secrets={filteredSecrets}
              onEdit={handleEdit}
              onDelete={handleDelete}
            />

          )}

        </Container>

      </Box>

      <SecretDialog
        open={dialogOpen}
        onClose={() => setDialogOpen(false)}
        onSave={handleSave}
        secret={selectedSecret}
        servers={servers}
      />

      <Snackbar
        open={snackbar.open}
        autoHideDuration={3000}
        onClose={() =>
          setSnackbar({
            ...snackbar,
            open: false,
          })
        }
      >
        <Alert
          severity={snackbar.severity}
        >
          {snackbar.message}
        </Alert>
      </Snackbar>

    </Box>

  );

}

export default SecretInventory;
