package com.vaultguardian.ai.service.impl;

import com.vaultguardian.ai.dto.AIRecommendation;
import com.vaultguardian.ai.enums.BreakGlassStatus;
import com.vaultguardian.ai.enums.Criticality;
import com.vaultguardian.ai.repository.BreakGlassRepository;
import com.vaultguardian.ai.repository.SecretRepository;
import com.vaultguardian.ai.repository.ServerRepository;
import com.vaultguardian.ai.service.AIRecommendationService;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Service;

import java.time.LocalDate;
import java.util.ArrayList;
import java.util.List;

@Service
@RequiredArgsConstructor
public class AIRecommendationServiceImpl
        implements AIRecommendationService {

    private final ServerRepository serverRepository;

    private final SecretRepository secretRepository;

    private final BreakGlassRepository breakGlassRepository;

    @Override
    public List<AIRecommendation> generateRecommendations() {

        List<AIRecommendation> list = new ArrayList<>();

        long expiringSecrets =
                secretRepository.countByExpiryDateBetween(
                        LocalDate.now(),
                        LocalDate.now().plusDays(30));

        if(expiringSecrets>0){

            list.add(
                    AIRecommendation.builder()
                            .severity("HIGH")
                            .category("Secret Rotation")
                            .title("Secrets Expiring Soon")
                            .recommendation(
                                    expiringSecrets+
                                            " secrets will expire within 30 days. Rotate immediately.")
                            .build());

        }

        long pending=
                breakGlassRepository.countByStatus(
                        BreakGlassStatus.PENDING);

        if(pending>0){

            list.add(
                    AIRecommendation.builder()
                            .severity("MEDIUM")
                            .category("BreakGlass")
                            .title("Pending Approval")
                            .recommendation(
                                    pending+
                                            " emergency access requests require approval.")
                            .build());

        }

        long critical=
                serverRepository.countByCriticality(
                        Criticality.CRITICAL);

        if(critical>2){

            list.add(
                    AIRecommendation.builder()
                            .severity("HIGH")
                            .category("Infrastructure")
                            .title("Critical Servers")
                            .recommendation(
                                    critical+
                                            " production servers are marked as CRITICAL. Review privileged access.")
                            .build());

        }

        long inactive=
                secretRepository.countByActive(false);

        if(inactive>0){

            list.add(
                    AIRecommendation.builder()
                            .severity("LOW")
                            .category("Secrets")
                            .title("Inactive Secrets")
                            .recommendation(
                                    inactive+
                                            " inactive secrets can be archived.")
                            .build());

        }

        if(list.isEmpty()){

            list.add(
                    AIRecommendation.builder()
                            .severity("INFO")
                            .category("Overall")
                            .title("Healthy Environment")
                            .recommendation(
                                    "No immediate security risks detected.")
                            .build());

        }

        return list;

    }

}
