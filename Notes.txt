@Override
public List<ServerResponse> getRecentServers() {

    return serverRepository
            .findTop5ByOrderByCreatedAtDesc()
            .stream()
            .map(server -> ServerResponse.builder()
                    .id(server.getId())
                    .hostname(server.getHostname())
                    .application(server.getApplication())
                    .environment(server.getEnvironment())
                    .status(server.getStatus())
                    .criticality(server.getCriticality())
                    .owner(server.getOwner())
                    .ipAddress(server.getIpAddress())
                    .build())
            .toList();

}
@Override
public List<SecretResponse> getExpiringSecrets() {

    return secretRepository
            .findByExpiryDateBeforeOrderByExpiryDateAsc(
                    LocalDate.now().plusDays(30))
            .stream()
            .map(secret -> SecretResponse.builder()
                    .id(secret.getId())
                    .secretName(secret.getSecretName())
                    .vaultReference(secret.getVaultReference())
                    .secretType(secret.getSecretType())
                    .rotationDays(secret.getRotationDays())
                    .lastRotated(secret.getLastRotated())
                    .expiryDate(secret.getExpiryDate())
                    .active(secret.getActive())
                    .serverName(secret.getServer().getHostname())
                    .build())
            .toList();

}
