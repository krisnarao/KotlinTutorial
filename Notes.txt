package com.vaultguardian.ai.entity;

import com.vaultguardian.ai.enums.UserRole;
import jakarta.persistence.*;
import lombok.*;

@Entity
@Table(name = "roles")
@Getter
@Setter
@NoArgsConstructor
@AllArgsConstructor
@Builder
public class Role extends BaseEntity {

    @Enumerated(EnumType.STRING)
    @Column(name = "role_name", unique = true, nullable = false)
    private UserRole roleName;

    @Column(nullable = false)
    private String description;

}

package com.vaultguardian.ai.repository;

import com.vaultguardian.ai.entity.Role;
import com.vaultguardian.ai.enums.UserRole;
import org.springframework.data.jpa.repository.JpaRepository;

import java.util.Optional;

public interface RoleRepository extends JpaRepository<Role, Long> {

    Optional<Role> findByRoleName(UserRole roleName);

}
