package com.vaultguardian.ai.service.impl;

import com.vaultguardian.ai.dto.DashboardResponse;
import com.vaultguardian.ai.enums.Criticality;
import com.vaultguardian.ai.repository.SecretRepository;
import com.vaultguardian.ai.repository.ServerRepository;
import com.vaultguardian.ai.service.DashboardService;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Service;

import java.time.LocalDate;

@Service
@RequiredArgsConstructor
public class DashboardServiceImpl implements DashboardService {

    private final ServerRepository serverRepository;

    private final SecretRepository secretRepository;

    @Override
    public DashboardResponse getDashboard() {

        return DashboardResponse.builder()

                .totalServers(serverRepository.count())

                .totalSecrets(secretRepository.count())

                .activeSecrets(secretRepository.countByActive(true))

                .inactiveSecrets(secretRepository.countByActive(false))

                .criticalServers(
                        serverRepository.countByCriticality(
                                Criticality.CRITICAL))

                .expiringSecrets(
                        secretRepository.countByExpiryDateBefore(
                                LocalDate.now().plusDays(30)))

                .build();

    }

}

package com.vaultguardian.ai.controller;

import com.vaultguardian.ai.dto.DashboardResponse;
import com.vaultguardian.ai.service.DashboardService;
import lombok.RequiredArgsConstructor;
import org.springframework.web.bind.annotation.*;

@RestController
@RequestMapping("/api/dashboard")
@RequiredArgsConstructor
public class DashboardController {

    private final DashboardService dashboardService;

    @GetMapping
    public DashboardResponse dashboard() {

        return dashboardService.getDashboard();

    }

}
