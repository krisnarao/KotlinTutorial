package com.vaultguardian.ai.service.impl;

import com.vaultguardian.ai.dto.SecretRequest;
import com.vaultguardian.ai.dto.SecretResponse;
import com.vaultguardian.ai.entity.Secret;
import com.vaultguardian.ai.entity.Server;
import com.vaultguardian.ai.repository.SecretRepository;
import com.vaultguardian.ai.repository.ServerRepository;
import com.vaultguardian.ai.service.SecretService;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Service;

import java.time.LocalDate;
import java.util.List;
import java.util.stream.Collectors;

@Service
@RequiredArgsConstructor
public class SecretServiceImpl implements SecretService {

    private final SecretRepository secretRepository;

    private final ServerRepository serverRepository;

    @Override
    public SecretResponse create(SecretRequest request) {

        if (secretRepository.existsBySecretName(request.getSecretName())) {
            throw new RuntimeException("Secret already exists");
        }

        Server server = serverRepository.findById(request.getServerId())
                .orElseThrow(() -> new RuntimeException("Server not found"));

        Secret secret = Secret.builder()
                .secretName(request.getSecretName())
                .vaultReference(request.getVaultReference())
                .secretType(request.getSecretType())
                .rotationDays(request.getRotationDays())
                .lastRotated(LocalDate.now())
                .expiryDate(LocalDate.now().plusDays(request.getRotationDays()))
                .active(true)
                .server(server)
                .build();

        return map(secretRepository.save(secret));
    }

    @Override
    public List<SecretResponse> findAll() {

        return secretRepository.findAll()
                .stream()
                .map(this::map)
                .collect(Collectors.toList());

    }

    @Override
    public SecretResponse findById(Long id) {

        Secret secret = secretRepository.findById(id)
                .orElseThrow(() -> new RuntimeException("Secret not found"));

        return map(secret);

    }

    @Override
    public SecretResponse update(Long id, SecretRequest request) {

        Secret secret = secretRepository.findById(id)
                .orElseThrow(() -> new RuntimeException("Secret not found"));

        Server server = serverRepository.findById(request.getServerId())
                .orElseThrow(() -> new RuntimeException("Server not found"));

        secret.setSecretName(request.getSecretName());
        secret.setVaultReference(request.getVaultReference());
        secret.setSecretType(request.getSecretType());
        secret.setRotationDays(request.getRotationDays());
        secret.setExpiryDate(LocalDate.now().plusDays(request.getRotationDays()));
        secret.setServer(server);

        return map(secretRepository.save(secret));

    }

    @Override
    public void delete(Long id) {

        secretRepository.deleteById(id);

    }

    private SecretResponse map(Secret secret) {

        return SecretResponse.builder()

                .id(secret.getId())

                .secretName(secret.getSecretName())

                .vaultReference(secret.getVaultReference())

                .secretType(secret.getSecretType())

                .rotationDays(secret.getRotationDays())

                .lastRotated(secret.getLastRotated())

                .expiryDate(secret.getExpiryDate())

                .active(secret.getActive())

                .serverName(secret.getServer().getHostname())

                .build();

    }

}

package com.vaultguardian.ai.controller;

import com.vaultguardian.ai.dto.SecretRequest;
import com.vaultguardian.ai.dto.SecretResponse;
import com.vaultguardian.ai.service.SecretService;
import jakarta.validation.Valid;
import lombok.RequiredArgsConstructor;
import org.springframework.web.bind.annotation.*;

import java.util.List;

@RestController
@RequestMapping("/api/secrets")
@RequiredArgsConstructor
public class SecretController {

    private final SecretService service;

    @PostMapping
    public SecretResponse create(
            @Valid
            @RequestBody
            SecretRequest request) {

        return service.create(request);

    }

    @GetMapping
    public List<SecretResponse> findAll() {

        return service.findAll();

    }

    @GetMapping("/{id}")
    public SecretResponse findById(
            @PathVariable Long id) {

        return service.findById(id);

    }

    @PutMapping("/{id}")
    public SecretResponse update(
            @PathVariable Long id,
            @RequestBody SecretRequest request) {

        return service.update(id, request);

    }

    @DeleteMapping("/{id}")
    public void delete(
            @PathVariable Long id) {

        service.delete(id);

    }

}
