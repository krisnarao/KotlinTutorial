package com.vaultguardian.ai.service.impl;

import com.vaultguardian.ai.dto.ServerRequest;
import com.vaultguardian.ai.dto.ServerResponse;
import com.vaultguardian.ai.entity.Server;
import com.vaultguardian.ai.enums.ServerStatus;
import com.vaultguardian.ai.repository.ServerRepository;
import com.vaultguardian.ai.service.ServerService;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Service;

import java.util.List;
import java.util.stream.Collectors;

@Service
@RequiredArgsConstructor
public class ServerServiceImpl implements ServerService {

    private final ServerRepository repository;

    @Override
    public ServerResponse create(ServerRequest request) {

        Server server = Server.builder()
                .hostname(request.getHostname())
                .application(request.getApplication())
                .environment(request.getEnvironment())
                .criticality(request.getCriticality())
                .owner(request.getOwner())
                .ipAddress(request.getIpAddress())
                .operatingSystem(request.getOperatingSystem())
                .location(request.getLocation())
                .status(ServerStatus.UP)
                .build();

        return map(repository.save(server));
    }

    @Override
    public List<ServerResponse> findAll() {

        return repository.findAll()
                .stream()
                .map(this::map)
                .collect(Collectors.toList());

    }

    @Override
    public ServerResponse findById(Long id) {

        Server server = repository.findById(id)
                .orElseThrow(() ->
                        new RuntimeException("Server not found"));

        return map(server);

    }

    @Override
    public ServerResponse update(Long id,
                                 ServerRequest request) {

        Server server = repository.findById(id)
                .orElseThrow(() ->
                        new RuntimeException("Server not found"));

        server.setHostname(request.getHostname());
        server.setApplication(request.getApplication());
        server.setEnvironment(request.getEnvironment());
        server.setCriticality(request.getCriticality());
        server.setOwner(request.getOwner());
        server.setIpAddress(request.getIpAddress());
        server.setOperatingSystem(request.getOperatingSystem());
        server.setLocation(request.getLocation());

        return map(repository.save(server));

    }

    @Override
    public void delete(Long id) {

        repository.deleteById(id);

    }

    private ServerResponse map(Server server) {

        return ServerResponse.builder()

                .id(server.getId())

                .hostname(server.getHostname())

                .application(server.getApplication())

                .environment(server.getEnvironment())

                .status(server.getStatus())

                .criticality(server.getCriticality())

                .owner(server.getOwner())

                .ipAddress(server.getIpAddress())

                .build();

    }

}

package com.vaultguardian.ai.controller;

import com.vaultguardian.ai.dto.ServerRequest;
import com.vaultguardian.ai.dto.ServerResponse;
import com.vaultguardian.ai.service.ServerService;
import jakarta.validation.Valid;
import lombok.RequiredArgsConstructor;
import org.springframework.web.bind.annotation.*;

import java.util.List;

@RestController
@RequestMapping("/api/servers")
@RequiredArgsConstructor
public class ServerController {

    private final ServerService service;

    @PostMapping
    public ServerResponse create(
            @Valid
            @RequestBody
            ServerRequest request) {

        return service.create(request);

    }

    @GetMapping
    public List<ServerResponse> getAll() {

        return service.findAll();

    }

    @GetMapping("/{id}")
    public ServerResponse getById(
            @PathVariable Long id) {

        return service.findById(id);

    }

    @PutMapping("/{id}")
    public ServerResponse update(
            @PathVariable Long id,
            @RequestBody ServerRequest request) {

        return service.update(id, request);

    }

    @DeleteMapping("/{id}")
    public void delete(
            @PathVariable Long id) {

        service.delete(id);

    }

}
