function getRiskColor(risk) {
  switch (risk) {
    case "LOW":
      return "success";
    case "MEDIUM":
      return "warning";
    case "HIGH":
      return "error";
    case "CRITICAL":
      return "error";
    default:
      return "default";
  }
}

function getStatusColor(status) {
  switch (status) {
    case "Pending":
      return "warning";
    case "Approved":
      return "success";
    case "Rejected":
      return "error";
    default:
      return "default";
  }
}

export const calculateRisk = (request) => {

    let score = 0;
    const reasons = [];
    const controls = [];

    // Environment
    if (request.environment === "PRODUCTION") {
        score += 30;
        reasons.push("Production environment");
        controls.push("Security approval required");
    } else {
        score += 5;
    }

    // Server Criticality
    switch (request.criticality) {

        case "CRITICAL":
            score += 30;
            reasons.push("Critical production server");
            controls.push("Enable session recording");
            break;

        case "HIGH":
            score += 20;
            reasons.push("High critical server");
            break;

        case "MEDIUM":
            score += 10;
            reasons.push("Medium critical server");
            break;

        default:
            score += 5;
    }

    // Secret Type
    if (request.secret?.includes("PASSWORD")) {

        score += 20;
        reasons.push("Privileged database password");

        controls.push("Rotate password after use");

    } else if (request.secret?.includes("KEY")) {

        score += 15;
        reasons.push("Sensitive API Key");

    } else if (request.secret?.includes("CERT")) {

        score += 10;
        reasons.push("Certificate access");

    }

    // Duration
    const duration = parseInt(request.duration);

    if (duration > 60) {

        score += 15;
        reasons.push("Long duration requested");

    } else {

        reasons.push("Access duration within policy");

    }

    // Incident
    if (request.incidentId) {

        score -= 5;
        reasons.push("Incident ID validated");

    }

    // Confidence

    let confidence = 95;

    if (score > 80)
        confidence = 98;

    if (score < 40)
        confidence = 92;

    // Risk

    let risk = "LOW";

    let recommendation = "AUTO APPROVE";

    if (score >= 80) {

        risk = "CRITICAL";

        recommendation = "SECURITY APPROVAL REQUIRED";

    }

    else if (score >= 60) {

        risk = "HIGH";

        recommendation = "MANAGER APPROVAL REQUIRED";

    }

    else if (score >= 40) {

        risk = "MEDIUM";

        recommendation = "MANUAL REVIEW";

    }

    // Business Impact

    const impact =
        request.environment === "PRODUCTION"
            ? "Customer facing production workload"
            : "Non Production workload";

    return {

        score,

        confidence,

        risk,

        recommendation,

        impact,

        reasons,

        controls,

        summary:
            `AI analysed ${request.server}. This request targets ${request.secret} on a ${request.environment} server. Risk level is ${risk}. ${recommendation}.`

    };

};
